scripted_effect declare_war_on_zulgurub_effect = {
	title:k_gurubashi.holder = { save_scope_as = zulgurub_holder }
	
	scope:kilrogg = {
		add_character_flag = {
			flag = free_mongol_cb
			days = 14
		}
		start_war = {
			cb = mongol_invasion_war
			target = title:k_gurubashi.holder
			target_title = title:k_gurubashi
		}
	}
}

## Start story cycle here
wc_bleeding_hollow_invasion.0001 = {
	type = character_event
	hidden = yes

	immediate = {
		debug_log = "The Bleeding Hollow clan invade Stranglethorn!"
		debug_log_date = yes

		root = {
			create_story = story_bleeding_hollow_invasion
			save_scope_as = story_owner
		}

		trigger_event = {
			id = wc_bleeding_hollow_invasion.1000
		}
	}
}
