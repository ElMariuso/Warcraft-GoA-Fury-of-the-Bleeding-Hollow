set_bleeding_hollow_characters_effect = {
	character:10005 = {
		save_scope_as = kilrogg
	}
}

try_to_set_bleeding_hollow_story_owner = {
	if = {
		limit = { $OWNER$ = { culture = { has_cultural_pillar = heritage_orcish } } }
		$OWNER$ = { add_character_modifier = { modifier = great_invader_modifier } }
		make_story_owner = $OWNER$
	}
	else = {
		end_story = yes
	}
}
